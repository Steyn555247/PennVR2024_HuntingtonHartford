0using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Custom_SceneOffTimer : MonoBehaviour
{
    public GameObject[] thingsToTurnOff;
    public float timeBetweenSteps;

    private float timeStamp;
    private float deltaTime;
    private int counter = 0;

    // Start is called before the first frame update
    void Start()
    {
        timeStamp = Time.time;

        for(int i = 0; i < thingsToTurnOff.Length;i++)
        {
            thingsToTurnOff[i].SetActive(true);
        }
    }

    // Update is called once per frame
    void Update()
    {
        if(Time.time - timeStamp > timeBetweenSteps)
        {
            //turn something on
            if (counter < thingsToTurnOff.Length)
            {
                thingsToTurnOff[counter].SetActive(false);
            }

            //stamp the current time
            timeStamp = Time.time;

            //add to count
            counter++;
        }
    }
}
